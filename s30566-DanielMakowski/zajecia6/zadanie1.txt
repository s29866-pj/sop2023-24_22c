#include <stdio.h>
#include <unistd.h>
#include <stdbool.h>
#include <signal.h>

bool wait = true;

void sigHandler(int sig){
    printf("Otrzymalem sygnal: %d\n", sig );
    wait = false;
}

int main(){
    signal(SIGUSR1, sigHandler);
    printf("PID: %d\n", getpid());
    
    while(wait){
        pause();
    }

    return 0;
}
