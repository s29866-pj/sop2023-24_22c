#!/bin/bash

n=$1

ciag(){
	local n=$1
	
	if [ "$n" -eq 0 ]; then
        	return 0
    	elif [ "$n" -eq 1 ]; then
        	return 1
    	fi

	ciag $((n - 1))
	local fib1="$?"
	ciag $((n - 2))
	local fib2="$?"

	return $((fib1 + fib2))
}

ciag $n
